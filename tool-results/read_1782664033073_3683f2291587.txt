    60→export class RuleEngine {
    61→  private rng: () => number;
    62→
    63→  constructor(seed = 12345) {
    64→    // PRNG แบบ deterministic (mulberry32) เพื่อ reproducibility
    65→    let s = seed >>> 0;
    66→    this.rng = () => {
    67→      s |= 0;
    68→      s = (s + 0x6d2b79f5) | 0;
    69→      let t = Math.imul(s ^ (s >>> 15), 1 | s);
    70→      t = (t + Math.imul(t ^ (t >>> 7), 61 | t)) ^ t;
    71→      return ((t ^ (t >>> 14)) >>> 0) / 4294967296;
    72→    };
    73→  }
    74→
    75→  pick<T>(arr: T[]): T {
    76→    return arr[Math.floor(this.rng() * arr.length)];
    77→  }
    78→
    79→  pickN<T>(arr: T[], n: number): T[] {
    80→    const copy = [...arr];
    81→    const out: T[] = [];
    82→    for (let i = 0; i < n && copy.length > 0; i++) {
    83→      const idx = Math.floor(this.rng() * copy.length);
    84→      out.push(copy.splice(idx, 1)[0]);
    85→    }
    86→    return out;
    87→  }
    88→
    89→  int(min: number, max: number): number {
    90→    return Math.floor(this.rng() * (max - min + 1)) + min;
    91→  }
    92→
    93→  randomVars(): TemplateVars {
    94→    return {
    95→      subject: this.pick(SUBJECTS),
    96→      topic: this.pick(TOPICS),
    97→      claim: this.pick(CLAIMS),
    98→      person: this.pick(SUBJECTS),
    99→      number: this.int(2, 99),
   100→      authority: this.pick(['ศาสตราจารย์', 'นักวิทยาศาสตร์', 'แพทย์ผู้เชี่ยวชาญ', 'นักเศรษฐศาสตร์', 'นักวิจัย']),
   101→      evidence: this.pick(['สถิติล่าสุด', 'รายงานการศึกษา', 'ผลสำรวจ', 'งานวิจัยใหม่', 'ข้อมูลจากแหล่งที่เชื่อถือ']),
   102→    };
   103→  }

   108→// AxiomCore Dataset Generator - Grammar Engine
   109→// เครื่องมือไวยากรณ์สำหรับสร้างประโยคไทยที่หลากหลาย (ไม่ใช้ LLM)
   110→
   111→
   112→// โครงสร้างไวยากรณ์พื้นฐานภาษาไทยสำหรับสร้างประโยคโต้แย้ง
   113→export type GrammarNode =
   114→  | { kind: 'literal'; text: string }
   115→  | { kind: 'var'; key: keyof TemplateVars }
   116→  | { kind: 'choice'; options: string[] }
   117→  | { kind: 'seq'; nodes: GrammarNode[] }
   118→  | { kind: 'optional'; node: GrammarNode; probability: number };
   119→
   120→export class GrammarEngine {
   121→  constructor(private rule: RuleEngine) {}
   122→
   123→  /**
   124→   สร้างประโยคจากโครงสร้างไวยากรณ์
   125→   */
   126→  generate(node: GrammarNode, vars: TemplateVars): string {
   127→    switch (node.kind) {
   128→      case 'literal':
   129→        return node.text;
   130→      case 'var':
   131→        return String(vars[node.key]);
   132→      case 'choice':
   133→        return this.rule.pick(node.options);
   134→      case 'seq':
   135→        return node.nodes.map((n) => this.generate(n, vars)).join('');
   136→      case 'optional':
   137→        return this.rule.int(0, 100) / 100 < node.probability
   138→          ? this.generate(node.node, vars)
   139→          : '';
   140→    }
   141→  }
   142→
   143→  /**
   144→   สร้างประโยคจาก template แบบง่าย ({var} หรือ {a|b|c})
   145→   */
   146→  template(tpl: string, vars: TemplateVars): string {
   147→    return tpl.replace(/\{([^}]+)\}/g, (_, expr: string) => {
   148→      // choice: a|b|c
   149→      if (expr.includes('|')) {
   150→        return this.rule.pick(expr.split('|'));
   151→      }
   152→      // variable
   153→      if (expr in vars) {
   154→        return String(vars[expr as keyof TemplateVars]);
   155→      }
   156→      return expr;
   157→    });
   158→  }
   159→}

   223→// AxiomCore Dataset Generator - Argument Template Engine
   224→// เครื่องมือเทมเพลตการโต้แย้งสำหรับสร้างตัวอย่างตรรกะวิบัติ 12 ประเภท 4 ระดับ (ไม่ใช้ LLM)

   227→export interface GeneratedExample {
   228→  text: string;
   229→  explanation: string;
   230→  reason: string;
   231→  label: string;
   232→}
   233→
   234→// เทมเพลตสำหรับแต่ละประเภทตรรกะวิบัติ แยกตามระดับความยาก
   235→// {subject} {topic} {claim} {person} {number} {authority} {evidence} เป็นตัวแปร
   236→// a|b|c เป็นทางเลือก
   237→type Template = (v: TemplateVars) => GeneratedExample;
   238→
   239→interface TemplateSet {
   240→  easy: Template[];
   241→  medium: Template[];
   242→  hard: Template[];
   243→  expert: Template[];
   244→}
   245→
   246→function tpl(t: string, explanation: string, reason: string, label: string): (v: TemplateVars) => GeneratedExample {
   247→  return (v) => ({ text: fill(t, v), explanation: fill(explanation, v), reason: fill(reason, v), label });
   248→}
   249→
   250→function fill(t: string, v: TemplateVars): string {
   251→  return t.replace(/\{([^}]+)\}/g, (_, expr: string) => {
   252→    if (expr.includes('|')) {
   253→      const opts = expr.split('|');
   254→      return opts[Math.floor(Math.random() * opts.length)];
   255→    }
   256→    return String(v[expr as keyof TemplateVars] ?? expr);
   257→  });
   258→}

   768→  generate(type: FallacyType, difficulty: Difficulty): GeneratedExample | null {
   769→    const set = TEMPLATES[type];

   776→  }

   790→export class DatasetGenerator {
   791→  private rule: RuleEngine;
   792→  private grammar: GrammarEngine;
   793→  private templates: ArgumentTemplateEngine;
   794→  private patterns: LogicPatternGenerator;
   795→
   796→  constructor(seed = 42) {
   797→    this.rule = new RuleEngine(seed);
   798→    this.grammar = new GrammarEngine(this.rule);
   799→    this.templates = new ArgumentTemplateEngine(this.rule);
   800→    this.patterns = new LogicPatternGenerator();
   801→  }
   802→
   803→  /**
   804→   สร้างตัวอย่างตรรกะวิบัติประเภทเดียวตามจำนวนที่ต้องการ
   805→   */
   806→  generateForType(type: FallacyType, count: number, difficulty?: Difficulty): FallacyExampleRecord[] {
   807→    const out: FallacyExampleRecord[] = [];
   808→    const difficulties: Difficulty[] = difficulty ? [difficulty] : ['easy', 'medium', 'hard', 'expert'];
   809→    let produced = 0;
   810→    let attempts = 0;
   811→    const maxAttempts = count * 10;
   812→    const seen = new Set<string>();
   813→
   814→    while (produced < count && attempts < maxAttempts) {
   815→      attempts++;
   816→      const diff = this.rule.pick(difficulties);
   817→      const ex = this.templates.generate(type, diff);
   818→      if (!ex) continue;
   819→      // หลีกเลี่ยงการซ้ำ
   820→      const key = `${type}:${diff}:${ex.text}`;
   821→      if (seen.has(key)) continue;
   822→      seen.add(key);
   823→      out.push({ ...ex, fallacyType: type, difficulty: diff });
   824→      produced++;
   825→    }
   826→    return out;
   827→  }
   828→
   829→  /**
   830→   สร้างตัวอย่างสำหรับทุกประเภทตรรกะวิบัติ (12 ประเภท)
   831→   */
   832→  generateAll(countPerType: number): FallacyExampleRecord[] {
   833→    const out: FallacyExampleRecord[] = [];
   834→    for (const type of ALL_FALLACY_TYPES) {
   835→      out.push(...this.generateForType(type, countPerType));
   836→    }
   837→    return out;
   838→  }
   839→
   840→  /**
   841→   สร้างตัวอย่างควบคุมระดับความยาก
   842→   */
   843→  generateByDifficulty(type: FallacyType, difficulty: Difficulty, count: number): FallacyExampleRecord[] {
   844→    return this.generateForType(type, count, difficulty);
   845→  }
   846→
   847→  getRuleEngine(): RuleEngine {
   848→    return this.rule;
   849→  }
   850→
   851→  getPatterns(): LogicPatternGenerator {
   852→    return this.patterns;
   853→  }
   854→}

   905→// เครื่องหมายเชื่อมประโยคเพื่อรวมตัวอย่างสองประเภท
   906→const CONNECTORS = [
   907→  ' นอกจากนี้ ',
   908→  ' และที่สำคัญกว่า ',
   909→  ' อีกอย่าง ',
   910→  ' ยิ่งไปกว่านั้น ',

   912→  ' ดังนั้น ',

   914→  ' เพราะฉะนั้น ',
   915→];

   917→export class MixedFallacyGenerator {
   918→  private rule: RuleEngine;
   919→  private templates: ArgumentTemplateEngine;
   920→
   921→  constructor(rule: RuleEngine) {
   922→    this.rule = rule;
   923→    this.templates = new ArgumentTemplateEngine(rule);
   924→  }
   925→
   926→  generate(count: number): MixedFallacyRecord[] {
   927→    const out: MixedFallacyRecord[] = [];
   928→    const seen = new Set<string>();
   929→    let attempts = 0;
   930→    const maxAttempts = count * 8;
   931→
   932→    while (out.length < count && attempts < maxAttempts) {
   933→      attempts++;
   934→      const [primary, secondary] = this.rule.pick(INTERESTING_PAIRS);
   935→      const ex1 = this.templates.generate(primary, this.rule.pick(['easy', 'medium', 'hard']));
   936→      const ex2 = this.templates.generate(secondary, this.rule.pick(['easy', 'medium', 'hard']));
   937→      if (!ex1 || !ex2) continue;
   938→      const connector = this.rule.pick(CONNECTORS);
   939→      const text = ex1.text + connector + ex2.text;
   940→      const key = `${primary}+${secondary}:${text}`;
   941→      if (seen.has(key)) continue;
   942→      seen.add(key);
   943→      out.push({
   944→        primaryType: primary,
   945→        secondaryType: secondary,
   946→        combination: `${primary}+${secondary}`,
   947→        text,
   948→        explanation: `ข้อความนี้ประกอบด้วยตรรกะวิบัติสองประเภท: ${FALLACY_META[primary].nameTh} และ ${FALLACY_META[secondary].nameTh} ${ex1.explanation} ส่วนต่อมา ${ex2.explanation}`,
   949→        reason: `ส่วนแรกเป็น ${FALLACY_META[primary].nameTh}: ${ex1.reason} ส่วนหลังเป็น ${FALLACY_META[secondary].nameTh}: ${ex2.reason}`,
   950→      });
   951→    }
   952→    return out;
   953→  }
   954→
   955→  /**
   956→   สร้างคู่ผสมเฉพาะ
   957→   */
   958→  generatePair(primary: FallacyType, secondary: FallacyType, count: number): MixedFallacyRecord[] {
   959→    const out: MixedFallacyRecord[] = [];
   960→    let attempts = 0;
   961→    const maxAttempts = count * 10;
   962→    while (out.length < count && attempts < maxAttempts) {
   963→      attempts++;
   964→      const ex1 = this.templates.generate(primary, this.rule.pick(['easy', 'medium']));
   965→      const ex2 = this.templates.generate(secondary, this.rule.pick(['easy', 'medium']));
   966→      if (!ex1 || !ex2) continue;
   967→      const connector = this.rule.pick(CONNECTORS);
   968→      out.push({
   969→        primaryType: primary,
   970→        secondaryType: secondary,
   971→        combination: `${primary}+${secondary}`,
   972→        text: ex1.text + connector + ex2.text,
   973→        explanation: `ผสม ${FALLACY_META[primary].nameTh} + ${FALLACY_META[secondary].nameTh}`,
   974→        reason: `${ex1.reason} และ ${ex2.reason}`,
   975→      });
   976→    }
   977→    return out;
   978→  }
   979→
   980→  /**
   981→   สรุปคู่ผสมที่เป็นไปได้ทั้งหมด
   982→   */
   983→  allCombinations(): [FallacyType, FallacyType][] {
   984→    const pairs: [FallacyType, FallacyType][] = [];
   985→    for (let i = 0; i < ALL_FALLACY_TYPES.length; i++) {
   986→      for (let j = 0; j < ALL_FALLACY_TYPES.length; j++) {
   987→        if (i !== j) pairs.push([ALL_FALLACY_TYPES[i], ALL_FALLACY_TYPES[j]]);
   988→      }
   989→    }
   990→    return pairs;
   991→  }

  1052→const SHOULD = ['ควร', 'ไม่ควร', 'จำเป็นต้อง', 'ไม่จำเป็นต้อง', 'สมควร', 'ไม่สมควร'];
  1053→
  1054→export class DebateGenerator {
  1055→  private rule: RuleEngine;
  1056→
  1057→  constructor(rule: RuleEngine) {
  1058→    this.rule = rule;
  1059→  }
  1060→
  1061→  /**
  1062→   สร้างหัวข้อโต้วาทีจำลองตามจำนวนที่ต้องการ
  1063→   */
  1064→  generate(count: number): DebateTopicRecord[] {
  1065→    const out: DebateTopicRecord[] = [];
  1066→    const categories = Object.keys(SUB_TOPICS) as DebateCategory[];
  1067→    const seen = new Set<string>();
  1068→    let attempts = 0;
  1069→    const maxAttempts = count * 5;
  1070→
  1071→    while (out.length < count && attempts < maxAttempts) {
  1072→      attempts++;
  1073→      const cat = this.rule.pick(categories);
  1074→      const sub = this.rule.pick(SUB_TOPICS[cat]);
  1075→      const pos = this.rule.pick(POSITIONS);
  1076→      const angle = this.rule.pick(ANGLES);
  1077→      const should = this.rule.pick(SHOULD);
  1078→
  1079→      // สร้างหัวข้อหลายรูปแบบ
  1080→      const format = this.rule.int(0, 4);
  1081→      let title = '';
  1082→      switch (format) {
  1083→        case 0:
  1084→          title = `${should}${pos}${sub}`;
  1085→          break;
  1086→        case 1:
  1087→          title = `${sub}: ${should}${pos}หรือไม่`;
  1088→          break;
  1089→        case 2:
  1090→          title = `${angle}: ${should}${pos}${sub}`;
  1091→          break;
  1092→        case 3:
  1093→          title = `${sub} — ${should}ถูก${pos}หรือไม่`;
  1094→          break;
  1095→        case 4:
  1096→          title = `การ${pos}${sub}${should === 'ควร' ? 'เป็นทางที่ถูกต้องหรือไม่' : 'ไม่เหมาะสมหรือไม่'}`;
  1097→          break;
  1098→      }
  1099→      if (seen.has(title)) continue;
  1100→      seen.add(title);
  1101→
  1102→      const premise = `ข้อโต้แย้ง: ${should}${pos}${sub} ${angle}`;
  1103→      out.push({ category: cat, title, premise });
  1104→    }
  1105→    return out;
  1106→  }
  1107→
  1108→  /**
  1109→   สร้างตามหมวด
  1110→   */
  1111→  generateForCategory(category: DebateCategory, count: number): DebateTopicRecord[] {
  1112→    const all = this.generate(count * 3);
  1113→    return all.filter((t) => t.category === category).slice(0, count);
  1114→  }
  1115→
  1116→  /**
  1117→   สรุปจำนวนหัวข้อย่อยในแต่ละหมวด
  1118→   */
  1119→  categoryStats(): Record<DebateCategory, number> {
  1120→    const stats = {} as Record<DebateCategory, number>;
  1121→    for (const cat of Object.keys(SUB_TOPICS) as DebateCategory[]) {
  1122→      stats[cat] = SUB_TOPICS[cat].length;
  1123→    }
  1124→    return stats;
  1125→  }
  1126→}
  1127→
  1128→export const ALL_DEBATE_CATEGORIES = Object.keys(SUB_TOPICS) as DebateCategory[];
  1129→
  1130→// ===== self-improvement.ts =====
  1131→// AxiomCore Self-Improvement Loop
  1132→// กระบวนการพัฒนาตนเอง: สร้าง -> วิเคราะห์ -> ตรวจสอบ -> หา FP/FN -> ปรับกฎ -> ทดสอบซ้ำ
  1133→
  1134→
  1135→export interface ImprovementIteration {
  1136→  iteration: number;
  1137→  totalTested: number;
  1138→  truePositive: number;
  1139→  falsePositive: number;
  1140→  falseNegative: number;
  1141→  trueNegative: number;
  1142→  precision: number;
  1143→  recall: number;
  1144→  f1Score: number;
  1145→  rulesAdjusted: number;
  1146→  notes: string;
  1147→}
  1148→
  1149→export interface SelfImprovementResult {
  1150→  iterations: ImprovementIteration[];
  1151→  finalPrecision: number;
  1152→  finalRecall: number;
  1153→  finalF1: number;
  1154→  totalTested: number;
  1155→  converged: boolean;
  1156→}
  1157→
  1158→// น้ำหนักกฎที่ปรับได้ในหน่วยความจำ (runtime) สำหรับ self-improvement
  1159→const ADJUSTABLE_WEIGHTS: Record<string, number> = {};
  1160→
  1161→export class SelfImprovementLoop {
  1162→  private analyzer = getAnalyzer();
  1163→  private generator: DatasetGenerator;
  1164→  private readonly minImprovement = 0.005; // นัยสำคัญของการปรับปรุง
  1165→  private readonly maxIterations = 5;
  1166→
  1167→  constructor(seed = 99) {
  1168→    this.generator = new DatasetGenerator(seed);
  1169→  }
  1170→
  1171→  /**
  1172→   ทำงานวนซ้ำจนกว่าจะไม่มีการปรับปรุงที่มีนัยสำคัญ หรือครบจำนวนรอบสูงสุด
  1173→   */
  1174→  async run(samplesPerType = 8): Promise<SelfImprovementResult> {
  1175→    const iterations: ImprovementIteration[] = [];
  1176→    let prevF1 = 0;
  1177→    let converged = false;
  1178→
  1179→    for (let it = 1; it <= this.maxIterations; it++) {
  1180→      // 1. สร้างตัวอย่างสำหรับทุกประเภท
  1181→      const examples = this.generateTestSet(samplesPerType);
  1182→
  1183→      // 2. วิเคราะห์และประเมิน
  1184→      const metrics = this.evaluate(examples);
  1185→
  1186→      // 3. ปรับกฎจาก FP/FN
  1187→      const rulesAdjusted = this.adjustRules(examples);
  1188→
  1189→      const iteration: ImprovementIteration = {
  1190→        iteration: it,
  1191→        totalTested: examples.length,
  1192→        ...metrics,
  1193→        rulesAdjusted,
  1194→        notes: this.buildNotes(it, metrics, rulesAdjusted),
  1195→      };
  1196→      iterations.push(iteration);
  1197→
  1198→      // 4. ตรวจสอบการลู่เข้า
  1199→      const improvement = metrics.f1Score - prevF1;
  1200→      if (it > 1 && improvement < this.minImprovement) {
  1201→        converged = true;
  1202→        break;
  1203→      }
  1204→      prevF1 = metrics.f1Score;
  1205→    }
  1206→
  1207→    const last = iterations[iterations.length - 1];
  1208→    return {
  1209→      iterations,
  1210→      finalPrecision: last.precision,
  1211→      finalRecall: last.recall,
  1212→      finalF1: last.f1Score,
  1213→      totalTested: iterations.reduce((s, i) => s + i.totalTested, 0),
  1214→      converged,
  1215→    };
  1216→  }
  1217→
  1218→  /**
  1219→   สร้างชุดทดสอบ: ตัวอย่างตรรกะวิบัติ + ตัวอย่างปกติ (เพื่อตรวจ FP)
  1220→   */
  1221→  private generateTestSet(samplesPerType: number): Array<FallacyExampleRecord & { expectedDetected: boolean }> {
  1222→    const set: Array<FallacyExampleRecord & { expectedDetected: boolean }> = [];
  1223→    // ตัวอย่างที่มีตรรกะวิบัติ (ควรถูกตรวจพบ)
  1224→    for (const type of ALL_FALLACY_TYPES) {
  1225→      const ex = this.generator.generateForType(type, samplesPerType);
  1226→      for (const e of ex) set.push({ ...e, expectedDetected: true });
  1227→    }
  1228→    // ตัวอย่างปกติที่ไม่มีตรรกะวิบัติ (เพื่อตรวจ FP) - สร้างประโยคธรรมดา
  1229→    const normals = [
  1230→      'การศึกษาเป็นรากฐานของการพัฒนาประเทศ',
  1231→      'เราควรออกกำลังกายเป็นประจำเพื่อสุขภาพที่ดี',
  1232→      'น้ำเป็นทรัพยากรที่จำเป็นสำหรับสิ่งมีชีวิตทุกชนิด',
  1233→      'การอ่านหนังสือช่วยพัฒนาความคิดและจินตนาการ',
  1234→      'เทคโนโลยีช่วยให้ชีวิตสะดวกสบายขึ้น',
  1235→      'การเคารพซึ่งกันและกันเป็นพื้นฐานของสังคมที่สันติ',
  1236→    ];
  1237→    for (const n of normals) {
  1238→      set.push({
  1239→        text: n,
  1240→        explanation: 'ประโยคปกติไม่มีตรรกะวิบัติ',
  1241→        reason: 'ไม่มีรูปแบบตรรกะวิบัติ',
  1242→        label: 'none',
  1243→        fallacyType: 'ad_hominem', // placeholder
  1244→        difficulty: 'easy',
  1245→        expectedDetected: false,
  1246→      });
  1247→    }
  1248→    return set;
  1249→  }
  1250→
  1251→  /**
  1252→   ประเมินผล: TP/FP/FN/TN
  1253→   */
  1254→  private evaluate(examples: Array<FallacyExampleRecord & { expectedDetected: boolean }>): {
  1255→    truePositive: number;
  1256→    falsePositive: number;
  1257→    falseNegative: number;
  1258→    trueNegative: number;
  1259→    precision: number;
  1260→    recall: number;
  1261→    f1Score: number;
  1262→  } {
  1263→    let tp = 0;
  1264→    let fp = 0;
  1265→    let fn = 0;
  1266→    let tn = 0;
  1267→
  1268→    for (const ex of examples) {
  1269→      const result = this.analyzer.analyze(ex.text);
  1270→      const detected = result.detectedFallacies;
  1271→      const detectedThisType = detected.some((d) => d.type === ex.fallacyType && d.confidence >= 0.4);
  1272→
  1273→      if (ex.expectedDetected) {
  1274→        // ควรถูกตรวจพบ
  1275→        if (detectedThisType) tp++;
  1276→        else fn++;
  1277→      } else {
  1278→        // ไม่ควรถูกตรวจพบ (ประโยคปกติ)
  1279→        if (detected.length > 0) fp++;
  1280→        else tn++;
  1281→      }
  1282→    }
  1283→
  1284→    const precision = tp + fp > 0 ? tp / (tp + fp) : 1;
  1285→    const recall = tp + fn > 0 ? tp / (tp + fn) : 1;
  1286→    const f1Score = precision + recall > 0 ? (2 * precision * recall) / (precision + recall) : 0;
  1287→
  1288→    return {
  1289→      truePositive: tp,
  1290→      falsePositive: fp,
  1291→      falseNegative: fn,
  1292→      trueNegative: tn,
  1293→      precision: Math.round(precision * 1000) / 1000,
  1294→      recall: Math.round(recall * 1000) / 1000,
  1295→      f1Score: Math.round(f1Score * 1000) / 1000,
  1296→    };
  1297→  }
  1298→
  1299→  /**
  1300→   ปรับกฎจากผล FP/FN (ปรับน้ำหนักในหน่วยความจำ)
  1301→   */
  1302→  private adjustRules(examples: Array<FallacyExampleRecord & { expectedDetected: boolean }>): number {
  1303→    let adjusted = 0;
  1304→    // หากมี FN สูงสำหรับประเภทหนึ่ง -> เพิ่มน้ำหนัก
  1305→    const fnByType = new Map<FallacyType, number>();
  1306→    for (const ex of examples) {
  1307→      if (!ex.expectedDetected) continue;
  1308→      const result = this.analyzer.analyze(ex.text);
  1309→      const detected = result.detectedFallacies.some((d) => d.type === ex.fallacyType && d.confidence >= 0.4);
  1310→      if (!detected) {
  1311→        fnByType.set(ex.fallacyType, (fnByType.get(ex.fallacyType) ?? 0) + 1);
  1312→      }
  1313→    }
  1314→    for (const [type, count] of fnByType) {
  1315→      if (count >= 2) {
  1316→        // เพิ่มน้ำหนักของกฎประเภทนี้เล็กน้อย
  1317→        ADJUSTABLE_WEIGHTS[type] = Math.min(0.15, (ADJUSTABLE_WEIGHTS[type] ?? 0) + 0.03);
  1318→        adjusted++;
  1319→      }
  1320→    }
  1321→    return adjusted;
  1322→  }
  1323→
  1324→  private buildNotes(it: number, metrics: { precision: number; recall: number; f1Score: number }, rulesAdjusted: number): string {
  1325→    return `รอบที่ ${it}: Precision ${Math.round(metrics.precision * 100)}%, Recall ${Math.round(metrics.recall * 100)}%, F1 ${Math.round(metrics.f1Score * 100)}%, ปรับกฎ ${rulesAdjusted} ข้อ`;
  1326→  }
  1327→}
  1328→
  1329→export function getAdjustedWeight(type: FallacyType): number {
  1330→  return ADJUSTABLE_WEIGHTS[type] ?? 0;
  1331→}
