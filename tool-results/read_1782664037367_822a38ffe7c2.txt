    60→    60→export class RuleEngine {
    61→    61→  private rng: () => number;
    62→    62→
    63→    63→  constructor(seed = 12345) {
    64→    64→    // PRNG แบบ deterministic (mulberry32) เพื่อ reproducibility
    65→    65→    let s = seed >>> 0;
    66→    66→    this.rng = () => {
    67→    67→      s |= 0;
    68→    68→      s = (s + 0x6d2b79f5) | 0;
    69→    69→      let t = Math.imul(s ^ (s >>> 15), 1 | s);
    70→    70→      t = (t + Math.imul(t ^ (t >>> 7), 61 | t)) ^ t;
    71→    71→      return ((t ^ (t >>> 14)) >>> 0) / 4294967296;
    72→    72→    };
    73→    73→  }
    74→    74→
    75→    75→  pick<T>(arr: T[]): T {
    76→    76→    return arr[Math.floor(this.rng() * arr.length)];
    77→    77→  }
    78→    78→
    79→    79→  pickN<T>(arr: T[], n: number): T[] {
    80→    80→    const copy = [...arr];
    81→    81→    const out: T[] = [];
    82→    82→    for (let i = 0; i < n && copy.length > 0; i++) {
    83→    83→      const idx = Math.floor(this.rng() * copy.length);
    84→    84→      out.push(copy.splice(idx, 1)[0]);
    85→    85→    }
    86→    86→    return out;
    87→    87→  }
    88→    88→
    89→    89→  int(min: number, max: number): number {
    90→    90→    return Math.floor(this.rng() * (max - min + 1)) + min;
    91→    91→  }

   109→   109→// เครื่องมือไวยากรณ์สำหรับสร้างประโยคไทยที่หลากหลาย (ไม่ใช้ LLM)

   112→   112→// โครงสร้างไวยากรณ์พื้นฐานภาษาไทยสำหรับสร้างประโยคโต้แย้ง
   113→   113→export type GrammarNode =
   114→   114→  | { kind: 'literal'; text: string }
   115→   115→  | { kind: 'var'; key: keyof TemplateVars }
   116→   116→  | { kind: 'choice'; options: string[] }
   117→   117→  | { kind: 'seq'; nodes: GrammarNode[] }
   118→   118→  | { kind: 'optional'; node: GrammarNode; probability: number };
   119→   119→
   120→   120→export class GrammarEngine {
   121→   121→  constructor(private rule: RuleEngine) {}
   122→   122→
   123→   123→  /**
   124→   124→   สร้างประโยคจากโครงสร้างไวยากรณ์
   125→   125→   */
   126→   126→  generate(node: GrammarNode, vars: TemplateVars): string {
   127→   127→    switch (node.kind) {
   128→   128→      case 'literal':
   129→   129→        return node.text;
   130→   130→      case 'var':
   131→   131→        return String(vars[node.key]);
   132→   132→      case 'choice':
   133→   133→        return this.rule.pick(node.options);
   134→   134→      case 'seq':
   135→   135→        return node.nodes.map((n) => this.generate(n, vars)).join('');
   136→   136→      case 'optional':
   137→   137→        return this.rule.int(0, 100) / 100 < node.probability
   138→   138→          ? this.generate(node.node, vars)
   139→   139→          : '';
   140→   140→    }
   141→   141→  }
   142→   142→
   143→   143→  /**
   144→   144→   สร้างประโยคจาก template แบบง่าย ({var} หรือ {a|b|c})
   145→   145→   */
   146→   146→  template(tpl: string, vars: TemplateVars): string {
   147→   147→    return tpl.replace(/\{([^}]+)\}/g, (_, expr: string) => {
   148→   148→      // choice: a|b|c
   149→   149→      if (expr.includes('|')) {
   150→   150→        return this.rule.pick(expr.split('|'));
   151→   151→      }
   152→   152→      // variable
   153→   153→      if (expr in vars) {
   154→   154→        return String(vars[expr as keyof TemplateVars]);
   155→   155→      }
   156→   156→      return expr;
   157→   157→    });
   158→   158→  }
   159→   159→}

   763→   763→
   764→   764→  constructor(rule: RuleEngine) {
   765→   765→    this.rule = rule;

   767→   767→
   768→   768→  generate(type: FallacyType, difficulty: Difficulty): GeneratedExample | null {
   769→   769→    const set = TEMPLATES[type];
   770→   770→    if (!set) return null;
   771→   771→    const list = set[difficulty];
   772→   772→    if (!list || list.length === 0) return null;

   790→   790→export class DatasetGenerator {
   791→   791→  private rule: RuleEngine;
   792→   792→  private grammar: GrammarEngine;
   793→   793→  private templates: ArgumentTemplateEngine;
   794→   794→  private patterns: LogicPatternGenerator;
   795→   795→
   796→   796→  constructor(seed = 42) {
   797→   797→    this.rule = new RuleEngine(seed);
   798→   798→    this.grammar = new GrammarEngine(this.rule);
   799→   799→    this.templates = new ArgumentTemplateEngine(this.rule);
   800→   800→    this.patterns = new LogicPatternGenerator();
   801→   801→  }
   802→   802→
   803→   803→  /**
   804→   804→   สร้างตัวอย่างตรรกะวิบัติประเภทเดียวตามจำนวนที่ต้องการ
   805→   805→   */
   806→   806→  generateForType(type: FallacyType, count: number, difficulty?: Difficulty): FallacyExampleRecord[] {
   807→   807→    const out: FallacyExampleRecord[] = [];
   808→   808→    const difficulties: Difficulty[] = difficulty ? [difficulty] : ['easy', 'medium', 'hard', 'expert'];
   809→   809→    let produced = 0;
   810→   810→    let attempts = 0;
   811→   811→    const maxAttempts = count * 10;
   812→   812→    const seen = new Set<string>();
   813→   813→
   814→   814→    while (produced < count && attempts < maxAttempts) {
   815→   815→      attempts++;
   816→   816→      const diff = this.rule.pick(difficulties);
   817→   817→      const ex = this.templates.generate(type, diff);
   818→   818→      if (!ex) continue;
   819→   819→      // หลีกเลี่ยงการซ้ำ
   820→   820→      const key = `${type}:${diff}:${ex.text}`;
   821→   821→      if (seen.has(key)) continue;
   822→   822→      seen.add(key);
   823→   823→      out.push({ ...ex, fallacyType: type, difficulty: diff });
   824→   824→      produced++;
   825→   825→    }
   826→   826→    return out;
   827→   827→  }
   828→   828→
   829→   829→  /**
   830→   830→   สร้างตัวอย่างสำหรับทุกประเภทตรรกะวิบัติ (12 ประเภท)
   831→   831→   */
   832→   832→  generateAll(countPerType: number): FallacyExampleRecord[] {
   833→   833→    const out: FallacyExampleRecord[] = [];
   834→   834→    for (const type of ALL_FALLACY_TYPES) {
   835→   835→      out.push(...this.generateForType(type, countPerType));
   836→   836→    }
   837→   837→    return out;
   838→   838→  }
   839→   839→
   840→   840→  /**
   841→   841→   สร้างตัวอย่างควบคุมระดับความยาก
   842→   842→   */
   843→   843→  generateByDifficulty(type: FallacyType, difficulty: Difficulty, count: number): FallacyExampleRecord[] {
   844→   844→    return this.generateForType(type, count, difficulty);
   845→   845→  }
   846→   846→
   847→   847→  getRuleEngine(): RuleEngine {
   848→   848→    return this.rule;
   849→   849→  }

   851→   851→  getPatterns(): LogicPatternGenerator {
   852→   852→    return this.patterns;
   853→   853→  }

  1267→  1267→
  1268→  1268→    for (const ex of examples) {
  1269→  1269→      const result = this.analyzer.analyze(ex.text);
  1270→  1270→      const detected = result.detectedFallacies;
  1271→  1271→      const detectedThisType = detected.some((d) => d.type === ex.fallacyType && d.confidence >= 0.4);
  1272→  1272→
  1273→  1273→      if (ex.expectedDetected) {
  1274→  1274→        // ควรถูกตรวจพบ
  1275→  1275→        if (detectedThisType) tp++;
  1276→  1276→        else fn++;
  1277→  1277→      } else {
  1278→  1278→        // ไม่ควรถูกตรวจพบ (ประโยคปกติ)
  1279→  1279→        if (detected.length > 0) fp++;
  1280→  1280→        else tn++;
  1281→  1281→      }
  1282→  1282→    }
  1283→  1283→
  1284→  1284→    const precision = tp + fp > 0 ? tp / (tp + fp) : 1;
  1285→  1285→    const recall = tp + fn > 0 ? tp / (tp + fn) : 1;
  1286→  1286→    const f1Score = precision + recall > 0 ? (2 * precision * recall) / (precision + recall) : 0;
  1287→  1287→
  1288→  1288→    return {
  1289→  1289→      truePositive: tp,
  1290→  1290→      falsePositive: fp,
  1291→  1291→      falseNegative: fn,
  1292→  1292→      trueNegative: tn,
  1293→  1293→      precision: Math.round(precision * 1000) / 1000,
  1294→  1294→      recall: Math.round(recall * 1000) / 1000,
  1295→  1295→      f1Score: Math.round(f1Score * 1000) / 1000,
  1296→  1296→    };
  1297→  1297→  }
  1298→  1298→
  1299→  1299→  /**
  1300→  1300→   ปรับกฎจากผล FP/FN (ปรับน้ำหนักในหน่วยความจำ)
  1301→  1301→   */
  1302→  1302→  private adjustRules(examples: Array<FallacyExampleRecord & { expectedDetected: boolean }>): number {
  1303→  1303→    let adjusted = 0;
  1304→  1304→    // หากมี FN สูงสำหรับประเภทหนึ่ง -> เพิ่มน้ำหนัก
  1305→  1305→    const fnByType = new Map<FallacyType, number>();
  1306→  1306→    for (const ex of examples) {
  1307→  1307→      if (!ex.expectedDetected) continue;
  1308→  1308→      const result = this.analyzer.analyze(ex.text);
  1309→  1309→      const detected = result.detectedFallacies.some((d) => d.type === ex.fallacyType && d.confidence >= 0.4);
  1310→  1310→      if (!detected) {
  1311→  1311→        fnByType.set(ex.fallacyType, (fnByType.get(ex.fallacyType) ?? 0) + 1);
  1312→  1312→      }
  1313→  1313→    }
  1314→  1314→    for (const [type, count] of fnByType) {
  1315→  1315→      if (count >= 2) {
  1316→  1316→        // เพิ่มน้ำหนักของกฎประเภทนี้เล็กน้อย
  1317→  1317→        ADJUSTABLE_WEIGHTS[type] = Math.min(0.15, (ADJUSTABLE_WEIGHTS[type] ?? 0) + 0.03);
  1318→  1318→        adjusted++;
  1319→  1319→      }
  1320→  1320→    }
  1321→  1321→    return adjusted;
  1322→  1322→  }
  1323→  1323→
  1324→  1324→  private buildNotes(it: number, metrics: { precision: number; recall: number; f1Score: number }, rulesAdjusted: number): string {
  1325→  1325→    return `รอบที่ ${it}: Precision ${Math.round(metrics.precision * 100)}%, Recall ${Math.round(metrics.recall * 100)}%, F1 ${Math.round(metrics.f1Score * 100)}%, ปรับกฎ ${rulesAdjusted} ข้อ`;
  1326→  1326→  }
  1327→  1327→}
  1328→  1328→

  1330→  1330→  return ADJUSTABLE_WEIGHTS[type] ?? 0;
